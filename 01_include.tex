\usepackage{tikz} % Drawing

% Xparse: Optional argument
\usepackage{xparse}
\makeatletter
\define@key{xarg}{shift}{\def\xshift{#1}}
\define@key{xarg}{scale}{\def\xscale{#1}}
\define@key{xarg}{rotate}{\def\xrotate{#1}}
\makeatother

%\DeclareExpandableDocumentCommand{\fishingfloat}{O{}}{%
\NewDocumentCommand{\fishingfloat}{O{}}{%
    % Fishing float
    \setkeys{xarg}{shift={(0,0)}, scale={1}, rotate={0},#1}%

    \begin{scope}[shift=\xshift, scale=\xscale, rotate=\xrotate]
    %\show{shift = \xshift}
    % Lines
    % Triangle
    \draw (0, 0) -- (1, 4);
    \draw (0, 0) -- (-1, 4);
    % 2 lines
    \draw (-1, 4) -- (1, 4);
    \draw (-0.9375, 3.75) -- (0.9375, 3.75);
    % Top bottom
    \draw (0, 5) -- (0, 5.5);
    \draw (0, 0) -- (0, -0.5);
    % Arc: start angle, end angle, radius
    \draw (1, 4) arc (0:180:1);
    \end{scope}
}
